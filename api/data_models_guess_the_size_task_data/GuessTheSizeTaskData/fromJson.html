static GuessTheSizeTaskData fromJson(json) {
  double newSize = json['correctSize'] as double;

  if (newSize <= 0) {
    throw Exception("Given size is not allowed");
  }

  double min = (0.33 + (0.33 * (Random().nextDouble()))) * newSize;
  double max = (1.25 + (0.75 * (Random().nextDouble()))) * newSize;

  double start = min + (max - min) * Random().nextDouble();
  print("Correct Size: " + newSize.toString() + ", start Size: " + start.toString() + ", min: " + min.toString() + ", max: " + max.toString());

  return GuessTheSizeTaskData(
    question: json["question"],
    questionImage: null,
    size: newSize,
    min: min,
    max: max,
    start: start,
  );
}
