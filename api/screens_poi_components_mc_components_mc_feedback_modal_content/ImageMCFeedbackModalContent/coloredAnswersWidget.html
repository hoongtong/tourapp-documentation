@override
Widget coloredAnswersWidget({context}) {
  List<Widget> gridImages = [];
  for (int i = 0; i < mcData.choices.length; i++) {
    // All correct ones in green and only if both wrong and selected then red
    Color borderColor;
    if (int.parse(answer) == i) {
      borderColor = (mcData.choices[i][1]) ? cubeGreen : cubeOrange;
    } else {
      borderColor = Colors.transparent;
    }

    gridImages.add(
      ClipRRect(
        borderRadius: BorderRadius.all(Radius.circular(10)),
        child: Container(
          color: borderColor,
          padding: EdgeInsets.all(6),
          child: Stack(
            children: [
              ClipRRect(
                borderRadius: BorderRadius.all(Radius.circular(7)),
                child: Image.file(
                  File(mcData.choices[i][0]),
                  height: double.infinity,
                  width: double.infinity,
                  fit: BoxFit.cover,
                ),
              ),
              Positioned(
                bottom: c1,
                right: c1,
                child: (mcData.choices[i][1]) ? Icon(Icons.check_circle_outline, color: cubeGreen) : Icon(Icons.cancel_outlined, color: cubeOrange),
              ),
            ],
          ),
        ),
      ),
    );
  }

  return GridView.count(
    physics: NeverScrollableScrollPhysics(),
    padding: EdgeInsets.fromLTRB(c1, 0, c1, c1),
    childAspectRatio: 1,
    shrinkWrap: true,
    crossAxisCount: 2,
    crossAxisSpacing: 10,
    mainAxisSpacing: 10,
    children: gridImages,
  );
}
