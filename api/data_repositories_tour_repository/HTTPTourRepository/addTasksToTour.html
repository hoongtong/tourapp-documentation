Future<Tour> addTasksToTour({Tour tour, TaskRepository taskRepository}) async {
  for (POI poi in tour.orderedPoiList) {
    POIData poiData = poi.poiData;
    // now we just take the first task but this can later easily be extended to accomodate more than 1 task per POI
    if (poiData.taskIDList.length > 0) {
      String taskID = poiData.taskIDList[0];
      poiData.task = await taskRepository.fetchTask(taskID);
    } else {
      poiData.task = Task(id: null, taskType: TaskType.NO_TASK, taskData: null);
    }
  }
  return tour;
}
