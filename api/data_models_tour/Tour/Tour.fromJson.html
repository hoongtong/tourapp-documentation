factory Tour.fromJson(Map<String, dynamic> json) {
  List<POI> pois = json["poiList"].map<POI>((json) => POI.fromJson(json)).toList();
  return Tour(
      id: json["id"],
      name: json["name"],
      description: json["description"],
      organisationID: json["organisationId"],
      organisationName: json["organisationName"],
      distanceMeters: (((json["distanceMeters"] as double).toInt() ~/ 100) * 100).toDouble(),
      orderedPoiList: pois);
}
