void StartDownload({BuildContext context, TourInfoScreenCubit tourInfoScreenCubit}) {
  showDialog<DownloadDialog>(
    context: context,
    builder: (_) => DownloadDialog(
        tourInfoScreenCubit: tourInfoScreenCubit,
        onContinue: () async {
          Navigator.pop(context);
          // User did not start prefetching
          if (tourInfoScreenCubit.state is TourInfoScreenLoaded) {
            HTTPTourRepository hTTPTourRepository = HTTPTourRepository();
            hTTPTourRepository.addModelsToTour(
                tour: tourInfoScreenCubit.tour, modelRepository: HTTPModelRepository(), tourInfoScreenCubit: tourInfoScreenCubit);
            hTTPTourRepository.downloadImagesOfTour(
                tour: tourInfoScreenCubit.tour, taskRepository: HTTPTaskRepository(), tourInfoScreenCubit: tourInfoScreenCubit);
          }
          StartTour(context: context, tourInfoScreenCubit: tourInfoScreenCubit);
        }),
  );
}
