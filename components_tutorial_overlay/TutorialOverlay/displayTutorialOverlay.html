static CoachMark displayTutorialOverlay(
    {@required String message,
    @required GlobalKey targetWidgetKey,
    @required double circleRadius,
    Function callback,
    Alignment textAlignment = Alignment.center}) {
  CoachMark coachMark = CoachMark();

  RenderBox target = targetWidgetKey.currentContext.findRenderObject();
  Rect markRect = target.localToGlobal(Offset.zero) & target.size;
  markRect = Rect.fromCircle(center: markRect.center, radius: circleRadius);

  coachMark.show(
    markRect: markRect,
    duration: null,
    targetContext: targetWidgetKey.currentContext,
    children: [
      Align(
        alignment: textAlignment,
        child: Padding(
          padding: const EdgeInsets.fromLTRB(0, 100, 0, 0),
          child: Column(
            children: [
              Text(
                message,
                style: CubeFonts.cubeSansSerif(color: Colors.white, fontSize: c5),
              ),
              SizedBox(height: c5),
              Row(
                mainAxisAlignment: MainAxisAlignment.center,
                children: [
                  Icon(
                    Icons.touch_app_rounded,
                    size: c3,
                    color: cubeWhite,
                  ),
                  Text(
                    "Tap the screen to continue.",
                    style: CubeFonts.cubeSansSerif(fontSize: c3, color: cubeWhite),
                  )
                ],
              )
            ],
          ),
        ),
      ),
    ],
    onClose: callback,
  );
  return coachMark;
}
