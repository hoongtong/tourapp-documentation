List<TourInfo> parseTourInfos(String responseBody) {
  // the actual tourlist in the Json is nested into _embedded and then into tours
  final parsed = jsonDecode(responseBody)["_embedded"]["tours"].cast<Map<String, dynamic>>();
  // shorthand notation for going through the list in Json and making TourInfo model from ever dict in the Json and saving it in a list locally
  return parsed.map<TourInfo>((Map<String, dynamic> json) => TourInfo.fromJson(json)).toList();
}
