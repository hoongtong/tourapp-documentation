@override
Widget coloredAnswersWidget({context}) {
  List<Widget> answerWidgets = [];
  for (int i = 0; i < mcData.choices.length; i++) {
    // All correct ones in green and only if both wrong and selected then red
    Color containerColor;
    if (int.parse(answer) == i) {
      containerColor = (mcData.choices[i][1]) ? cubeGreen : cubeOrange;
    } else {
      containerColor = Colors.transparent;
    }

    answerWidgets.add(
      Padding(
        padding: const EdgeInsets.only(bottom: c4),
        child: Container(
          decoration: BoxDecoration(
            color: cubeWhite,
            boxShadow: [cubeBoxShadow],
            borderRadius: BorderRadius.all(Radius.circular(20)),
            border: Border.all(color: containerColor, width: 3),
          ),
          width: MediaQuery.of(context).size.width * 0.6,
          child: Padding(
            padding: const EdgeInsets.symmetric(vertical: c1 + 3),
            child: Row(
              children: [
                SizedBox(width: c2),
                (mcData.choices[i][1]) ? Icon(Icons.check_circle_outline, color: cubeGreen) : Icon(Icons.cancel_outlined, color: cubeOrange),
                SizedBox(width: c2),
                Flexible(
                  child: Text(
                    mcData.choices[i][0],
                    textAlign: TextAlign.left,
                  ),
                ),
                SizedBox(width: c2),
              ],
            ),
          ),
        ),
      ),
    );
  }
  return Column(
    children: answerWidgets,
  );
}
