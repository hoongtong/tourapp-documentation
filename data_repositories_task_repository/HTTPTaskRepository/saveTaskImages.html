Future<void> saveTaskImages(task) async {
  if (task.taskType == TaskType.MC) {
    MCData mcData = task.taskData;
    // Question image
    if (mcData.mcType == MCType.QUESTION_IMAGE || mcData.mcType == MCType.BOTH_IMAGE) {
      if (!mcData.questionImage.contains('/')) mcData.questionImage = await downloadImage(mcData.questionImage);
    }
    // Answer image
    if (mcData.mcType == MCType.ANSWER_IMAGE || mcData.mcType == MCType.BOTH_IMAGE) {
      for (int i = 0; i < mcData.choices.length; i++) {
        if (!mcData.choices[i][0].contains('/')) mcData.choices[i][0] = await downloadImage(mcData.choices[i][0]);
      }
    }
  }
  // 2) Detective Task
  if (task.taskType == TaskType.DETECTIVE) {
    DetectiveTaskData detectiveTaskData = task.taskData;
    if (detectiveTaskData.questionImage != null && !detectiveTaskData.questionImage.contains("/"))
      detectiveTaskData.questionImage = await downloadImage(detectiveTaskData.questionImage);
  }
}
